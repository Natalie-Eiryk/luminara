class MetacognitiveEngine {
  constructor(persistence) {
    this.persistence = persistence;
    this.errorPatterns = this.loadErrorPatterns();
  }

  analyzeError(question, selectedAnswer, correctAnswer) {
    const errorType = this.classifyError(question, selectedAnswer, correctAnswer);
    const relatedConcepts = this.findRelatedConcepts(question);
    const strategy = this.recommendStrategy(errorType);
    const reflection = this.generateReflectionPrompt(errorType, question);

    // Store for Misconception Dungeon
    this.recordMisconception({
      questionId: question.id,
      errorType,
      timestamp: Date.now(),
      resolved: false
    });

    return { errorType, relatedConcepts, strategy, reflection };
  }

  classifyError(q, selected, correct) {
    // Analyze answer similarity
    const similarity = this.calculateSimilarity(
      q.options[selected],
      q.options[correct]
    );

    if (similarity > 0.7) return 'CONFUSION'; // Similar terms
    if (this.wasRecentlyCorrect(q.id)) return 'CARELESS';
    if (this.isTerminologyQuestion(q)) return 'VOCABULARY';
    if (this.requiresReasoning(q)) return 'LOGIC';
    return 'MEMORY';
  }

  generateReflectionPrompt(errorType, question) {
    const prompts = {
      CONFUSION: `These concepts are similar. What's the KEY difference between ${this.getConfusedTerms(question)}?`,
      MEMORY: `Let's strengthen this memory. Can you create a mnemonic for this?`,
      LOGIC: `Walk through the reasoning step by step. What rule applies here?`,
      CARELESS: `You knew this! Slow down and read carefully next time.`,
      VOCABULARY: `Term confusion detected. Let's define both terms clearly.`
    };
    return prompts[errorType];
  }

  getUnresolvedMisconceptions() {
    return this.persistence.data.misconceptions?.filter(m => !m.resolved) || [];
  }
}

class MisconceptionDungeon {
  constructor(metacognition, battleScene) {
    this.meta = metacognition;
    this.battle = battleScene;
    this.currentFloor = 0;
    this.monsters = [];
  }

  generateDungeon() {
    const misconceptions = this.meta.getUnresolvedMisconceptions();

    // Group by error type
    const floors = this.groupIntoFloors(misconceptions);

    return {
      floors: floors.map((floor, i) => ({
        number: i + 1,
        theme: floor.errorType,
        monsters: floor.misconceptions.map(m => this.createMonster(m)),
        boss: i === floors.length - 1 ? this.createBoss(floor) : null
      })),
      totalMonsters: misconceptions.length,
      difficulty: this.calculateDifficulty(misconceptions)
    };
  }

  createMonster(misconception) {
    const templates = {
      CONFUSION: {
        name: 'Confusion Golem',
        sprite: 'golem',
        color: '#fbbf24',
        ability: 'Presents both confused terms'
      },
      MEMORY: {
        name: 'Memory Phantom',
        sprite: 'phantom',
        color: '#a855f7',
        ability: 'Fades in/out, requires recall'
      },
      LOGIC: {
        name: 'Logic Demon',
        sprite: 'demon',
        color: '#ef4444',
        ability: 'Multi-step reasoning required'
      },
      VOCABULARY: {
        name: 'Vocabulary Specter',
        sprite: 'specter',
        color: '#3b82f6',
        ability: 'Definition matching battle'
      },
      CARELESS: {
        name: 'Careless Imp',
        sprite: 'imp',
        color: '#22c55e',
        ability: 'Same question, careful mode'
      }
    };

    const template = templates[misconception.errorType];
    return {
      ...template,
      misconception,
      hp: this.calculateMonsterHP(misconception),
      defeatCondition: this.getDefeatCondition(misconception)
    };
  }

  defeatMonster(monster, wasCorrect) {
    if (wasCorrect) {
      // Mark misconception progress
      monster.misconception.correctStreak =
        (monster.misconception.correctStreak || 0) + 1;

      if (monster.misconception.correctStreak >= 3) {
        monster.misconception.resolved = true;
        this.meta.persistence.save();
        return { defeated: true, reward: this.generateReward(monster) };
      }

      return { defeated: false, progress: monster.misconception.correctStreak };
    }

    // Wrong - monster counterattacks
    monster.misconception.correctStreak = 0;
    return { defeated: false, counterattack: true };
  }
}

class CreatureSystem {
  constructor(persistence) {
    this.persistence = persistence;
    this.creatures = this.loadCreatures();
  }

  creatureTemplates = {
    '000': {
      name: 'Cell Sprite',
      stages: ['Mitochip', 'Cytofly', 'Celldragon', 'NUCLEON'],
      colors: ['#a7f3d0', '#34d399', '#10b981', '#047857'],
      abilities: [null, 'hint_organelle', 'buff_biology', 'mastery_cellular']
    },
    '100': {
      name: 'Brain Wisp',
      stages: ['Synapseed', 'Neurowing', 'Cortexian', 'CEREBRIX'],
      colors: ['#ddd6fe', '#a78bfa', '#7c3aed', '#4c1d95'],
      abilities: [null, 'hint_pathway', 'buff_memory', 'mastery_neural']
    },
    // ... other categories
  };

  getCreatureState(categoryId) {
    const mastery = this.persistence.getCategoryMastery(categoryId);
    const template = this.creatureTemplates[categoryId];

    const stage = this.calculateStage(mastery);
    const happiness = this.calculateHappiness(categoryId);
    const hunger = this.calculateHunger(categoryId);

    return {
      categoryId,
      name: template.stages[stage] || template.name,
      stage,
      mastery,
      happiness,
      hunger,
      color: template.colors[stage],
      ability: template.abilities[stage],
      canEvolve: this.checkEvolutionReady(categoryId, stage),
      lastFed: this.persistence.data.creatures?.[categoryId]?.lastFed
    };
  }

  calculateStage(mastery) {
    if (mastery >= 90) return 3; // Legendary
    if (mastery >= 75) return 2; // Adult
    if (mastery >= 50) return 1; // Juvenile
    if (mastery >= 25) return 0; // Larva
    return -1; // Egg
  }

  feed(categoryId) {
    // Called when answering questions in this category
    const creature = this.persistence.data.creatures[categoryId] || {};
    creature.lastFed = Date.now();
    creature.feedCount = (creature.feedCount || 0) + 1;
    this.persistence.data.creatures[categoryId] = creature;
    this.persistence.save();

    return this.getCreatureState(categoryId);
  }

  checkBonding(cat1, cat2) {
    // Related categories can "bond" for mutual benefits
    const relations = {
      '000': ['100', '400'], // Cells relate to Nervous, Tissue
      '100': ['000', '200', '500'], // Nervous relates to Cells, Sensory, Organs
      // ...
    };

    return relations[cat1]?.includes(cat2);
  }
}

class RhythmEngine {
  constructor(bpm = 90) {
    this.bpm = bpm;
    this.beatInterval = 60000 / bpm;
    this.currentBeat = 0;
    this.isPlaying = false;
    this.callbacks = [];
  }

  start() {
    this.isPlaying = true;
    this.lastBeatTime = performance.now();
    this.tick();
  }

  tick() {
    if (!this.isPlaying) return;

    const now = performance.now();
    const elapsed = now - this.lastBeatTime;

    if (elapsed >= this.beatInterval) {
      this.currentBeat++;
      this.lastBeatTime = now;
      this.callbacks.forEach(cb => cb(this.currentBeat));
    }

    requestAnimationFrame(() => this.tick());
  }

  getTimingWindow() {
    const now = performance.now();
    const timeSinceBeat = now - this.lastBeatTime;
    const timeToNextBeat = this.beatInterval - timeSinceBeat;

    // Distance to nearest beat (0-1, 0 = on beat)
    const distanceToBeat = Math.min(timeSinceBeat, timeToNextBeat) / (this.beatInterval / 2);

    if (distanceToBeat < 0.1) return 'PERFECT';
    if (distanceToBeat < 0.2) return 'GREAT';
    if (distanceToBeat < 0.4) return 'GOOD';
    return 'MISS';
  }

  getMultiplier(timing) {
    return { PERFECT: 2.0, GREAT: 1.5, GOOD: 1.0, MISS: 0.5 }[timing];
  }
}

class TemporalHeist {
  constructor(questionPool, persistence) {
    this.questions = questionPool;
    this.persistence = persistence;
    this.paradoxMeter = 0;
    this.currentEra = 0;
    this.rewinds = 3;
  }

  eras = [
    { name: 'Ancient', difficulty: 0.3, theme: 'stone', color: '#a1887f' },
    { name: 'Classical', difficulty: 0.5, theme: 'marble', color: '#90a4ae' },
    { name: 'Modern', difficulty: 0.7, theme: 'steel', color: '#78909c' },
    { name: 'Future', difficulty: 0.85, theme: 'neon', color: '#00bcd4' },
    { name: 'Quantum', difficulty: 0.95, theme: 'void', color: '#7c4dff' }
  ];

  startHeist() {
    this.paradoxMeter = 0;
    this.currentEra = 0;
    this.rewinds = 3;
    this.securityGates = this.generateGates();

    return this.getCurrentState();
  }

  generateGates() {
    return this.eras.map(era => ({
      era: era.name,
      questions: this.selectQuestionsForDifficulty(era.difficulty, 5),
      completed: 0,
      required: 5
    }));
  }

  answerQuestion(answer, isCorrect) {
    if (isCorrect) {
      this.securityGates[this.currentEra].completed++;

      if (this.securityGates[this.currentEra].completed >=
          this.securityGates[this.currentEra].required) {
        return this.advanceEra();
      }

      return { success: true, gateProgress: this.getGateProgress() };
    }

    // Wrong answer = paradox
    this.paradoxMeter++;

    if (this.paradoxMeter >= 10) {
      return { timelineCollapse: true, finalScore: this.calculateScore() };
    }

    return {
      success: false,
      paradox: this.paradoxMeter,
      effect: this.getParadoxEffect()
    };
  }

  useRewind() {
    if (this.rewinds <= 0) return false;
    this.rewinds--;
    this.paradoxMeter = Math.max(0, this.paradoxMeter - 1);
    return true;
  }

  getParadoxEffect() {
    // Higher paradox = harder questions
    const effects = [
      'Minor temporal distortion',
      'Difficulty increased',
      'Timeline becoming unstable',
      'Causality breaking down',
      'CRITICAL: Timeline near collapse!'
    ];
    return effects[Math.min(Math.floor(this.paradoxMeter / 2), 4)];
  }
}

class ConstellationSystem {
  constructor(persistence) {
    this.persistence = persistence;
  }

  getConstellationData(categoryId) {
    const mastery = this.persistence.getCategoryMastery(categoryId);
    const concepts = this.persistence.getConceptsMastered(categoryId);

    return {
      categoryId,
      name: this.getConstellationName(categoryId),
      stars: this.generateStars(categoryId, concepts),
      connections: this.generateConnections(concepts),
      completionPercent: mastery,
      unlocked: mastery >= 25,
      animated: mastery >= 90
    };
  }

  generateStars(categoryId, concepts) {
    // Position stars in recognizable pattern based on category
    const patterns = {
      '000': this.cellPattern(),
      '100': this.brainPattern(),
      '200': this.eyePattern(),

    const pattern = patterns[categoryId] || this.defaultPattern();

    return concepts.map((concept, i) => ({
      id: concept.id,
      x: pattern[i % pattern.length].x,
      y: pattern[i % pattern.length].y,
      lit: concept.mastered,
      brightness: concept.mastery / 100,
      name: concept.name

  }

    const categories = this.persistence.getAllCategories();
